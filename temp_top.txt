function IsTickTypeAvailable(const AType: TTickType): Boolean;
begin
  case AType of
    ttTSC: Result := IsTSCAvailable;
  else
    // 鏍囧噯/绯荤粺/楂樼簿搴?鏈€浣筹細鍦ㄥ悇骞冲彴鍧囧彲閫€鍖栨彁渚?
    Result := True;
  end;
end;

function GetTickTypeName(const AType: TTickType): string;

end;

function BestTickClock: TTickClock;

  Result := TTickClock.Best;

function GetAvailableTickTypes: TTickTypeArray;
var
  i: TTickType;
  list: array of TTickType;
  count: Integer;
begin
  SetLength(list, Ord(High(TTickType)) + 1);
  count := 0;
  for i := Low(TTickType) to High(TTickType) do
  begin
    if IsTickTypeAvailable(i) then
    begin
      list[count] := i;
      Inc(count);
    end;
  end;
  SetLength(list, count);
  Result := list;
end;

// === TTickClock 瀹炵幇 ===

class function TTickClock.Best: TTickClock;
begin
  Result.FSource := ttBest;
end;

class function TTickClock.FromType(const Src: TTickType): TTickClock;
begin
  Result.FSource := Src;
end;

function TTickClock.Now: UInt64;
{$IFDEF MSWINDOWS}
var counter: Int64;
{$ELSE}
var ts: timespec;
    tv: timeval;
{$ENDIF}
begin
  case FSource of
    ttTSC:
      Result := ReadTSCTick;
    {$IFDEF MSWINDOWS}
    ttHighPrecision, ttBest:
      begin
        QueryPerformanceCounter(counter);
        Result := UInt64(counter);
      end;
    ttStandard, ttSystem:
      Result := GetTickCount64;
    {$ELSE}
    ttHighPrecision, ttBest:
      begin
        clock_gettime(CLOCK_MONOTONIC, @ts);
        Result := UInt64(ts.tv_sec) * 1000000000 + UInt64(ts.tv_nsec);
      end;
    ttStandard, ttSystem:
      begin
        fpgettimeofday(@tv, nil);
        Result := UInt64(tv.tv_sec) * 1000000 + UInt64(tv.tv_usec);
      end;
    {$ENDIF}
  else
    Result := 0;
  end;
end;

function TTickClock.FrequencyHz: UInt64;
{$IFDEF MSWINDOWS}
var freq: Int64;
{$ENDIF}
begin
  case FSource of
    ttTSC:
      Result := GetTSCFrequency;
    {$IFDEF MSWINDOWS}
    ttHighPrecision, ttBest:
      begin
        if QueryPerformanceFrequency(freq) then Result := UInt64(freq) else Result := 0;
      end;
    ttStandard, ttSystem:
      Result := 1000;
    {$ELSE}
    ttHighPrecision, ttBest:
      Result := 1000000000;
    ttStandard, ttSystem:
      Result := 1000000;
    {$ENDIF}
  else
    Result := 0;
  end;
end;

function TTickClock.Elapsed(const Start: UInt64): UInt64;
begin
  Result := Now - Start;
end;

function TTickClock.MinStep: TDuration;
begin
  Result := TicksToDuration(1);
end;

function TTickClock.TicksToDuration(const T: UInt64): TDuration;
{$IFDEF MSWINDOWS}
var freq: Int64;
{$ENDIF}
var f: UInt64;
begin
  case FSource of
    ttTSC:
      begin
        f := GetTSCFrequency;
        if f = 0 then Exit(TDuration.Zero);
        Result := TDuration.FromNs((T * 1000000000) div f);
      end;
    {$IFDEF MSWINDOWS}
    ttHighPrecision, ttBest:
      begin
        if not QueryPerformanceFrequency(freq) then Exit(TDuration.Zero);
        Result := TDuration.FromNs((T * 1000000000) div UInt64(freq));
      end;
    ttStandard, ttSystem:
      Result := TDuration.FromMs(T);
    {$ELSE}
    ttHighPrecision, ttBest:
      Result := TDuration.FromNs(T);
    ttStandard, ttSystem:
      Result := TDuration.FromUs(T);
    {$ENDIF}
  else
    Result := TDuration.Zero;
  end;
end;

function TTickClock.DurationToTicks(const D: TDuration): UInt64;
{$IFDEF MSWINDOWS}
var freq: Int64;
{$ENDIF}
var f: UInt64;
begin
  case FSource of
    ttTSC:
      begin
        f := GetTSCFrequency;
        if f = 0 then Exit(0);
        Result := (D.AsNs * f) div 1000000000;
      end;
    {$IFDEF MSWINDOWS}
    ttHighPrecision, ttBest:
      begin
        if not QueryPerformanceFrequency(freq) then Exit(0);
        Result := (D.AsNs * UInt64(freq)) div 1000000000;
      end;
    ttStandard, ttSystem:
      Result := D.AsMs;
    {$ELSE}
    ttHighPrecision, ttBest:
      Result := D.AsNs;
    ttStandard, ttSystem:
      Result := D.AsUs;
    {$ENDIF}
  else
    Result := 0;
  end;
end;

function TTickClock.IsMonotonic: Boolean;
begin
  case FSource of
    ttTSC: Result := IsTSCAvailable;
    {$IFDEF MSWINDOWS}
    ttHighPrecision, ttBest: Result := True;
    ttStandard, ttSystem:   Result := True; // GetTickCount64 鍗曡皟閫掑
    {$ELSE}
    ttHighPrecision, ttBest: Result := True;
    ttStandard, ttSystem:   Result := False; // gettimeofday 闈炲崟璋?
    {$ENDIF}
  else
    Result := False;
  end;
end;



function QuickMeasureClock(const AProc: TProc; const Clock: TTickClock): TDuration;
var
  start: UInt64;
begin
  start := Clock.Now;
  AProc();
  Result := Clock.TicksToDuration(Clock.Elapsed(start));
end;

// 蹇€熸祴閲忓嚱鏁帮紙榛樿浣跨敤 BestTickClock锛?
function QuickMeasure(const AProc: TProc): TDuration;
var
  C: TTickClock;
begin
  C := TTickClock.Best;
  Result := QuickMeasureClock(AProc, C);
end;
